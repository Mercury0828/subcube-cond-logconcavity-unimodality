import random, math, numpy as np
from numpy.polynomial import polynomial as P

def roots_real_nonpos(coeff, tol=1e-8):
    # coeff ascending for polynomial sum coeff[j] t"'^j
    # trim
    coeff=np.array(coeff,dtype=float)
    if np.all(np.abs(coeff)<1e-12): return True
    while len(coeff)>1 and abs(coeff[-1])<1e-12: coeff=coeff[:-1]
    r=np.roots(coeff[::-1])
    return all(abs(z.imag)<1e-6 and z.real<=1e-6 for z in r)

def short_ok(a,k):
    n=len(a)-1
    for m in range(1,k+1):
        binom=[math.comb(m,j) for j in range(m+1)]
        for u in range(n-m+1):
            coeff=[binom[j]*a[u+j] for j in range(m+1)]
            if not roots_real_nonpos(coeff):
                return False,(m,u,coeff,np.roots(np.array(coeff[::-1])))
    return True,None

def global_rr(a):
    n=len(a)-1
    coeff=[math.comb(n,j)*a[j] for j in range(n+1)]
    return roots_real_nonpos(coeff)

def logconcave(a):
    return all(a[i]*a[i] >= a[i-1]*a[i+1]-1e-9 for i in range(1,len(a)-1))

for n in [8,10,12,16,20]:
  for k in [3,4,5]:

      slopes=[]
      s=random.uniform(-1,1)

    print('none',n,k)
print('done')
